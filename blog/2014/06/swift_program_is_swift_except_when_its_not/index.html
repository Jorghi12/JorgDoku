Swift Program is Swift, Except When it’s Not

Date Wed, 11 Jun 2014 Tags Apple / Programming Language / Benchmarks / Bubble Sort

This past WWDC Apple revealed a brand new programming language called Swift to the masses and passed it out to Apple Developers packaged in the XCode 6 Beta.

Language Features

  • Type Inferencing and Automatic Reference Counting (ARC)
  • Optional Bindings and Generic Functions
  • Integration with Objective-C
  • First-Class Functions and Closures

Modern, Safe and Powerful? Choose Two

In an earlier post, I decided to write a fun bubble sort as an algorithmic "Hello World". But there was only once issue, It was super slow.

Swift Version

import Cocoa

func bubbleSort(list: CLong[]) -> CLong[]
{
    let length = list.count
    var count: CLong, pos: CLong

    for (var count = 0; count < length-1; count++) {
        for (var pos = 0; pos < (length-count-1); pos++) {
            if (list[pos] > list[pos+1]) {
                (list[pos], list[pos+1]) =
                    (list[pos+1], list[pos])
            }
        }
    }
    return list
}

func sortMain(problem_size: CLong) -> CLong[]
{
    var array: CLong[] = []

    srandomdev()
    for _ in 0..problem_size {
        array += (random() % problem_size)
    }
    return bubbleSort(array)
}

C Version

#include <stdio.h>
#include <stdlib.h>
#include <time.h>

void bubble_sort(long list[], long n)
{
    long count, pos, temp;
    for (count = 0; count < (n-1); count++) {
        for (pos = 0; pos < (n-count-1); pos++) {
            if (list[pos] > list[pos+1])
            {
                temp        =   list[pos];
                list[pos]   =   list[pos+1];
                list[pos+1] =   temp;
            }
        }
    }
}

int main(int argc, char *argv[])
{
    long problem_size = strtoul(argv[1], NULL, 10);
    long array[problem_size];
    long count;

    srand(time(0));
    for (count = 0; count < problem_size; count++) {
        array[count] = rand() % problem_size;
    }

    bubble_sort(array, problem_size);

    for (count = 0; count < problem_size; count++) {
        printf("%ld\n", array[count]);
    }
    return 0;
}

CPython Version

import os, sys

from random import randint

def bubble_sort(a_list):
    length = len(a_list)
    for count in range(0, length):
        for pos in range(0, length-count-1):
            if a_list[pos] > a_list[pos + 1]:
                a_list[pos], a_list[pos+1] = a_list[pos+1], a_list[pos]
    return a_list

def main(args):
    if len(args)-1 == 1:
        problem_size = int(args[1])
    else:
        problem_size = 10000
    array = [randint(0, problem_size) for num in range(0, problem_size)]

    return bubble_sort(array)


if __name__ == '__main__':
    sys.exit(main(sys.argv))
