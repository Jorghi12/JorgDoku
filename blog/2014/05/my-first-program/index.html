Bourne Again Programmer

Date Thu, 29 May 2014

I have a love hate relationship with the unix terminal. For many who write software the shell is a dark abyss of programs past and write off it’s presence as a relic of tradition. For others, the terminal is a gate into a small quiet piece of patch of space they own to hang their tools and work in peace.

  1 #!/usr/bin/env bash
  2 usage() { cat <<- DOCUMENT
  3 
  4     $PROGNAME [-h] [-t] [-k key] [-l login] [server]
  5 
  6     Utility for automatically exporting ssh keys into remote servers for authentication.
  7     Checks $HOME/.ssh for id_rsa.pub or id_dsa.pub and appends them into
  8     the authorized_keys of remote host
  9 
 10     AUTHOR:     Sang Han
 11     YEAR:       2013
 12     VERSION:    2.3
 13 
 14     -h [help]
 15         Outputs usage directions
 16     -t [test]
 17         Runs unit tests
 18     -k [key]
 19         Specify your own public key located in $HOME/.ssh directory
 20     -l [login]
 21         Specify user login credentials
 22 
 23 
 24     DOCUMENT
 25     exit 0
 26 }
 27 
 28 # Global Variables
 29 PROGNAME="$(basename "$0")"
 30 
 31 function error_exit() {
 32     printf "ERROR due to %s"  "${1:-"Unknown Error"}"
 33     exit 1
 34 }
 35 
 36 function test_names() {
 37     printf "\$KEY is %s\n" "${KEY[i]}"
 38     printf "\$PUBKEY is %s\n" "${PUBKEY[i]}"
 39     printf "\$LOGIN_USER is %s\n" "${LOGIN_USER}"
 40     printf "\$SERVER is %s\n" "${SERVER}"
 41     printf "\n"
 42 }
 43 
 44 function install_keys() {
 45     # Reads the public key file with cat and then logs into ssh server
 46     # and appends the public key file at the end of the authorized_key file
 47     # within $HOME
 48     cat < "${PUBKEY[i]}" | \
 49         ssh -l "$LOGIN_USER" "$SERVER" "\
 50             if [ -d ~/.ssh ]; then \
 51                 cat >> ~/.ssh/authorized_keys; \
 52             else \
 53                 mkdir ~/.ssh; \
 54                 cat >> ~/.ssh/authorized_keys; \
 55             fi"
 56 }
 57 
 58 function check_keys() {
 59     # Checks $HOME/.ssh directory for default keys
 60     if [[ $HOME/.ssh/*.pub ]]; then
 61         return
 62     else
 63         error_exit "missing ssh public keys"
 64     fi
 65 }
 66 
 67 function check_auth() {
 68     # Checks if proper credentials have been given as $1. After credential has
 69     # been checked, will read and bind the variable.
 70     local CREDENTIAL=$1
 71     if [[ ! ${!CREDENTIAL} ]]; then
 72         read -p "${CREDENTIAL}:  " "${CREDENTIAL}"
 73     fi
 74 }
 75 
 76 function check_server() {
 77     SERVER=$1
 78     if [[ "$SERVER" =~ @ ]]; then
 79         LOGIN_USER=${SERVER%@*}
 80         SERVER=${SERVER#*@}
 81         return
 82     fi
 83 }
 84 
 85 main() {
 86     # Declare default key files to find if user does not specify
 87     if [[ -z $KEY ]]; then declare -a KEY=("id_rsa" "id_dsa"); fi
 88 
 89     for (( i=0; i<${#KEY[@]}; i+=1 )); do
 90         # Path to key file
 91         local PUBKEY[i]=$HOME/.ssh/${KEY[i]}.pub
 92 
 93         # Unit Test runs every iteration
 94         if ((TEST==1)); then
 95             test_names
 96             return
 97         fi
 98 
 99         # Gather necessary login credentials
100         if [[ -r "${PUBKEY[i]}" ]]; then
101             check_auth LOGIN_USER
102             check_auth SERVER
103             install_keys && break
104         fi
105     done
106 }
107 
108 # Option Parsing
109 declare -i TEST=0
110 while getopts ":k:l:th" OPTION; do
111     case $OPTION in
112         h) usage
113             ;;
114         k) KEY=$OPTARG
115              ;;
116         t) TEST=1
117               ;;
118         l) LOGIN_USER=$OPTARG
119              ;;
120         ?) { echo "Invalid option: -${OPTARG}"; usage; } >&2
121              exit 1
122              ;;
123     esac
124 done
125     shift $((OPTIND-1))
126 
127 if [[ "$0" == "${BASH_SOURCE}" ]]; then
128 
129     # Validate user input
130     if [[ $# = 1 ]]; then check_server "$1"; fi
131     check_keys
132 
133     main
134 fi
